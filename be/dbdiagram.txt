Table users{
  id int [pk]
  avatar varchar
  email varchar
  password varchar
  name varchar
  role varchar
  phone varchar
  created_at datetime
  updated_at datatime
}

Table categories {
  id int [pk]
  parent_id int [null]
  name varchar
  description varchar
  created_at datatime
  updated_at datatime
}

Table wishlist {
  id int [pk]
  user_id int
  product_id int
  created_at datetime
  updated_at datetime
}

Ref: wishlist.user_id <> users.id
Ref: wishlist.product_id <> products.id

Table products{
  id int [pk]
  name varchar
  price int
  oldprice int
  image text
  description text
  specification text
  buyturn int
  quantity int
  brand_id int
  category_id int
  created_at datetime
  updated_at datatime
}

Ref: categories.id < products.category_id

Table brands{
  id int [pk]
  name varchar
  image text
}

Ref: brands.id < products.brand_id

Table feedback{
  id int [pk]
  product_id int
  user_id int
  star int
  content text
  created_at datetime
  updated_at datatime
}

Ref: feedback.product_id > products.id
Ref: feedback.user_id > users.id

Table address{
  id int [pk]
  user_id int
  commune varchar
  district varchar
  city varchar
  created_at datetime
  updated_at datatime
}

Ref: address.user_id > users.id

Table orders{
  id int [pk]
  user_id int
  payment_id int
  address_id int
  status int
  note text
  totalPrice int
  created_at datetime
  updated_at datatime
}

Ref: orders.user_id > users.id
Ref: orders.address_id - address.id

Table order_details{
  id int [pk]
  order_id int
  product_id int
  price int
  quantity int
  created_at datetime
  updated_at datatime
}

Ref: orders.id < order_details.order_id
Ref: products.id < order_details.product_id

Table posts{
  id int [pk]
  user_id int
  title text
  image text
  content text
  created_at datetime
  updated_at datatime
}

Ref: posts.user_id > users.id


Table payment {
  id int [pk]
  order_id int
  amount int
  type varchar
  status varchar
  created_at datetime
  updated_at datetime
}

Ref: payment.order_id - orders.id
Ref: payment.id - orders.payment_id

Table cart {
  id int [pk]
  user_id int
  total int
  created_at datetime
  updated_at datetime
}

Table cart_item {
  id int [pk]
  cart_id int
  product_id int
  quantity int
  created_at datetime
  updated_at datetime
}

Ref: cart.user_id - users.id
Ref: cart_item.cart_id > cart.id
Ref: cart_item.product_id <> products.id
